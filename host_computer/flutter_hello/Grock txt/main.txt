  // 单位换算系数
  double _getUnitFactor(String unit) {
    switch (unit) {
      case 'kHz':
        return 1e3;
      case 'MHz':
        return 1e6;
      case 'GHz':
        return 1e9;
      default: // Hz
        return 1.0;
    }
  }

  // 解析频率值为Hz（考虑单位）
  double? _parseFreq(String text, String unit) {
    final double? value = double.tryParse(text);
    if (value == null || value < 0) return null;
    return value * _getUnitFactor(unit);
  }

  // 格式化频率值（自动选择最优单位并保留两位小数）
  String _formatFreqAutoUnit(double freqHz, [int decimalPlaces = 2]) {
    if (freqHz >= 1e9) {
      return '${(freqHz / 1e9).toStringAsFixed(decimalPlaces)} GHz';
    } else if (freqHz >= 1e6) {
      return '${(freqHz / 1e6).toStringAsFixed(decimalPlaces)} MHz';
    } else if (freqHz >= 1e3) {
      return '${(freqHz / 1e3).toStringAsFixed(decimalPlaces)} kHz';
    } else {
      return '${freqHz.toStringAsFixed(decimalPlaces)} Hz';
    }
  }

  // 格式化频率值（指定单位并保留两位小数）
  String _formatFreq(double freqHz, String unit, [int decimalPlaces = 2]) {
    final double factor = _getUnitFactor(unit);
    return '${(freqHz / factor).toStringAsFixed(decimalPlaces)}';
  }

  // 弹窗提醒
  void _showErrorDialog(String message) {
    showDialog(
      context: context,
      builder: (context) => ContentDialog(
        title: const Text('参数错误'),
        content: Text(message),
        actions: [
          Button(
            child: const Text('确定'),
            onPressed: () => Navigator.pop(context),
          ),
        ],
      ),
    );
  }

  // 从起始/终止频率更新中心/跨度
  void _updateFreqFromStartStop() {
    final double? startHz = _parseFreq(startFreqController.text, startFreqUnit.value);
    final double? stopHz = _parseFreq(stopFreqController.text, stopFreqUnit.value);

    // 合法性校验
    if (startHz == null || stopHz == null) {
      _showErrorDialog('起始/终止频率输入无效，请输入数字');
      return;
    }
    if (startHz >= stopHz) {
      _showErrorDialog('起始频率不能大于等于终止频率');
      return;
    }

    // 计算中心和跨度（Hz）
    final double centerHz = (startHz + stopHz) / 2;
    final double spanHz = stopHz - startHz;

    // 更新控制器和单位（自动选择最优单位）
    setState(() {
      // 中心频率
      centerFreqController.text = _formatFreq(centerHz, centerFreqUnit.value);
      // 跨度
      spanController.text = _formatFreq(spanHz, spanUnit.value);
    });
  }

  // 从中心/跨度更新起始/终止频率
  void _updateFreqFromCenterSpan() {
    final double? centerHz = _parseFreq(centerFreqController.text, centerFreqUnit.value);
    final double? spanHz = _parseFreq(spanController.text, spanUnit.value);

    // 合法性校验
    if (centerHz == null || spanHz == null) {
      _showErrorDialog('中心频率/扫宽输入无效，请输入数字');
      return;
    }
    if (spanHz <= 0) {
      _showErrorDialog('扫宽必须大于0');
      return;
    }

    // 计算起始/终止（Hz）
    final double startHz = centerHz - spanHz / 2;
    final double stopHz = centerHz + spanHz / 2;

    // 校验起始频率非负
    if (startHz < 0) {
      _showErrorDialog('计算出的起始频率为负数，请调整中心频率或扫宽');
      return;
    }

    // 更新控制器和单位
    setState(() {
      startFreqController.text = _formatFreq(startHz, startFreqUnit.value);
      stopFreqController.text = _formatFreq(stopHz, stopFreqUnit.value);
    });
  }

    // 移除首尾空格
    controller.text = controller.text.trim();
    // 触发计算
    calculateFunc();
  }

  // 构建带单位下拉的输入行（支持回车触发计算）
  Widget _buildInputRow({
    required String label,
    required TextEditingController controller,
    required ValueNotifier<String> unitNotifier,
    required List<String> units,
    VoidCallback? onSubmitted,
  }) {
    return Row(
      children: [
        SizedBox(
          width: 100,
          child: Text(label, style: const TextStyle(color: material.Colors.white)),
        ),
        Expanded(
          child: TextBox(
            controller: controller,
            // 回车时触发
            onSubmitted: (value) => onSubmitted?.call(),
          ),
        ),
        const SizedBox(width: 8),
        SizedBox(
          width: 80,
          child: ValueListenableBuilder<String>(
            valueListenable: unitNotifier,
            builder: (context, value, child) {
              return ComboBox<String>(
                value: value,
                items: units.map((unit) => ComboBoxItem<String>(value: unit, child: Text(unit))).toList(),
                onChanged: (newValue) {
                  if (newValue != null) {
                    unitNotifier.value = newValue;
                    // 单位切换时触发计算
                    if (onSubmitted != null) {
                      onSubmitted();
                    }
                  }
                },
              );
            },
          ),
        ),
      ],
    );
  }

  // 生成频谱示例数据
  static List<FlSpot> generateSpectrumData({
    required int pointCount,
    required double minFreq,
    required double maxFreq,
    required double minDbm,
    required double maxDbm,
  }) {
    final math.Random random = math.Random();
    final double freqStep = (maxFreq - minFreq) / (pointCount - 1);
    return List.generate(pointCount, (index) {
      final double freq = minFreq + index * freqStep;
      final double dbm = minDbm + random.nextDouble() * (maxDbm - minDbm);
      return FlSpot(freq, dbm);
    });
  }
